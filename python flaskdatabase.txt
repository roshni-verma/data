from flask import Flask,request,redirect,url_for,render_template #flask - package, Flask -module/class
from flask_mysqldb import MySQL

app =Flask(__name__)    #app is the object , __name__ variable/argument

#configuaration of database
app.config['MYSQL_HOST'] = 'localhost'
#app.config['MYSQL_PORT'] = 3307
app.config['MYSQL_USER'] = 'root'
app.config['MYSQL_PASSWORD'] = ''
app.config['MYSQL_DB'] = 'flaskDB'

mysql = MySQL(app)


@app.route('/')

@app.route('/login', methods=['POST','GET'])
def home():
    if request.method =='POST':
        username =request.form['uname']
        userplace =request.form['uplace'] #username = $_POST['uname]-  php 
        #creating a connection
        cur = mysql.connection.cursor()
        #cur.execute('''create database flaskDB''')
        #cur.execute('''create table logindetails(name varchar(50), place varchar(50));''')
        cur.execute('''insert into logindetails(name,place) values(%s,%s);''',(username,userplace))
        #saving the actions performed 
        mysql.connection.commit()
        cur.close() #close the connection
        return 'Successfully inserted data'
